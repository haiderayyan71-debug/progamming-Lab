#include <iostream>
using namespace std;

 main( )
 {
    string name;
    float matric, intermediate, ecat, aggregate;

    
    cout << "Enter the student's name: ";
    cin >> name ;
    cout << "Enter matriculation marks (out of 1100): ";
    cin >> matric;
    cout << "Enter intermediate marks (out of 550): ";
    cin >> intermediate;
    cout << "Enter ECAT Marks (out of 400): ";
    cin >> ecat;

    Calculation based on weightage: 10% Matric, 40% Inter, 50% ECAT
    aggregate = (matric / 1100 * 10) + (intermediate / 550 * 40) + (ecat / 400 * 50);

     cout << "Aggregate score for " << name << " is " << aggregate << "%" << endl;

    return 0;
}
